function animateIn(el, delay=0){
  gsap.set(el,{ opacity:0, y:12, scale:0.995, rotate:gsap.utils.random(-2,3), transformOrigin:"50% 50%" });
  gsap.to(el,{ opacity:1, y:0, scale:1, duration:0.42, ease:EASE, delay });
  // idle: flotaci√≥n muy sutil
  gsap.to(el,{ y:"+=6", duration:3.2, yoyo:true, repeat:-1, ease:"sine.inOut", delay:delay+0.42 });
  gsap.to(el,{ rotate:`+=${gsap.utils.random(-0.6,0.6)}`, duration:6, yoyo:true, repeat:-1, ease:"sine.inOut", delay:delay+0.42 });
}
function animateOut(el, cb){
  gsap.to(el,{ opacity:0, y:-12, scale:.995, filter:"blur(2px)", duration:0.28, ease:EASE, onComplete:cb });
}

const stack = document.getElementById('stack');
const MAX_VISIBLE = 4;
const queue = [];

function pushToast(node){
  queue.push(node);
  processQueue();
}
function processQueue(){
  while (stack.children.length < MAX_VISIBLE && queue.length){
    const el = queue.shift();
    stack.appendChild(el);
    animateIn(el, 0.06 * stack.children.length); // stagger
    const hold = 2400 + Math.random()*800;
    setTimeout(()=>animateOut(el, ()=>{ el.remove(); processQueue(); }), hold);
  }
}
